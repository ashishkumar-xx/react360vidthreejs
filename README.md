# Install

    npm install --save react360vidthree

# Usage

    import React, { Component } from 'react'
    import Video360Three from 'react360vidthree'
     class Example extends Component {
       render () {
           return (<Video360Three
                    Id="video1"
                    autoRotate="true"
                    src="https://bit.ly/2YIgYqw"
                    width="480"
                    height="270"/>
                    )}
                  }
